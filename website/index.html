快速开始

安装 
mkdir my-m7s-server && cd my-m7s-server
go mod init my-m7s-server
创建主程序 
package main

import (
    "context"
    "m7s.live/v5"
    _ "m7s.live/v5/plugin/rtmp"
    _ "m7s.live/v5/plugin/flv"
    _ "m7s.live/v5/plugin/hls"
)

func main() {
    m7s.Run(context.Background(), "config.yaml")
}
